\subsection*{Text}
To display text in center with a 3 sec wait time.
\begin{minted}{python}
from manim import *

class WriteText(Scene):
    def construct(self):
        text = Text("This is a just a text")
        self.play(Write(text))
        self.wait(3)
  \end{minted}
